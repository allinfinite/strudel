// Voice-following synthesizer
note(audioInNote)
  .s("sine")
  .gain(audioInEnvelope.range(0.5, 1))
  .lpf(audioInCentroid.range(500, 5000))
  // Add harmony based on volume
  .stack(
    note(audioInNote.add(7))
      .s("triangle")
      .gain(audioInVolume.mul(0.4)),
    note(audioInNote.sub(12))
      .s("sawtooth")
      .gain(audioInVolume.mul(0.3))
  )
